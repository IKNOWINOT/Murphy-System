        <div class="terminal-container">
            <div class="terminal" id="terminal"></div>
            <div class="input-area">
                <button onclick="testConnection()" id="testBtn">TEST CONNECTION</button>
                <button onclick="initializeSystem()" id="initBtn" disabled>INITIALIZE SYSTEM</button>
                <button onclick="getStatus()" id="statusBtn" disabled>GET STATUS</button>

    <script>
        // Use relative path since we're on same domain
        const API_BASE = '/api';
        let connected = false;
        let currentStates = [];

        function terminalOutput(tag, message) {
            const terminal = document.getElementById('terminal');
            const line = document.createElement('div');
            line.className = 'terminal-line';
            
            const tagSpan = document.createElement('span');
            tagSpan.className = `tag tag-${tag}`;
            tagSpan.textContent = tag.toUpperCase();
            
            line.appendChild(tagSpan);
            line.appendChild(document.createTextNode(message));
            
            terminal.appendChild(line);
            terminal.scrollTop = terminal.scrollHeight;
        }

        async function testConnection() {
            terminalOutput('system', 'Testing connection to Murphy System backend...');
            
            try {
                // Use relative URL since we're on same server
                const response = await fetch('/api/status');
                const data = await response.json();
                
                connected = true;
                document.getElementById('status').textContent = 'CONNECTED';
                document.getElementById('status').style.background = '#00ff41';
                
                terminalOutput('success', 'Connection successful!');
                terminalOutput('api', `Phase: ${data.phase}`);
                terminalOutput('api', `Initialized: ${data.initialized}`);
                terminalOutput('api', `States: ${data.states_count}`);
                
                // Enable buttons
                document.getElementById('initBtn').disabled = false;
                document.getElementById('statusBtn').disabled = false;
                document.getElementById('statesBtn').disabled = false;
                document.getElementById('evolveBtn').disabled = false;
                
            } catch (error) {
                terminalOutput('error', `Connection failed: ${error.message}`);
                terminalOutput('system', 'Make sure backend is running on port 8888');
                document.getElementById('status').textContent = 'DISCONNECTED';
                document.getElementById('status').style.background = '#ff4444';
            }
        }

        async function initializeSystem() {
            if (!connected) {
                terminalOutput('error', 'Not connected to backend');
                return;
            }

            terminalOutput('system', 'Initializing Murphy System...');
            document.getElementById('initBtn').disabled = true;
            
            try {
                const response = await fetch('/api/initialize', {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                });
                const data = await response.json();
                
                terminalOutput('success', 'System initialized successfully!');
                terminalOutput('api', `Root state: ${data.root_state.id} - ${data.root_state.name}`);
                terminalOutput('api', `Confidence: ${data.root_state.confidence.toFixed(2)}`);
                terminalOutput('api', `States created: ${data.status.states_count}`);
                terminalOutput('api', `Gates active: ${data.status.gates_count}`);
                terminalOutput('api', `Swarms running: ${data.status.swarms_count}`);
                terminalOutput('api', `Constraints: ${data.status.constraints_count}`);
                
            } catch (error) {
                terminalOutput('error', `Initialization failed: ${error.message}`);
                document.getElementById('initBtn').disabled = false;
            }
        }

        async function getStatus() {
            if (!connected) {
                terminalOutput('error', 'Not connected to backend');
                return;
            }

            terminalOutput('system', 'Fetching system status...');
            
            try {
                const response = await fetch('/api/status');
                const data = await response.json();
                
                terminalOutput('success', 'Status retrieved:');
                terminalOutput('api', `Phase: ${data.phase}`);
                terminalOutput('api', `Confidence: ${data.confidence.toFixed(2)}`);
                terminalOutput('api', `Murphy Index: ${data.murphy_index.toFixed(2)}`);
                terminalOutput('api', `States: ${data.states_count}`);
                terminalOutput('api', `Artifacts: ${data.artifacts_count}`);
                terminalOutput('api', `Active Gates: ${data.gates_count}`);
                terminalOutput('api', `Active Swarms: ${data.swarms_count}`);
                terminalOutput('api', `Constraints: ${data.constraints_count}`);
                
            } catch (error) {
                terminalOutput('error', `Failed to get status: ${error.message}`);
            }
        }

        async function getStates() {
            if (!connected) {
                terminalOutput('error', 'Not connected to backend');
                return;
            }

            terminalOutput('system', 'Fetching all states...');
            
            try {
                const response = await fetch('/api/states');
                const data = await response.json();
                currentStates = data;
                
                terminalOutput('success', `Found ${data.length} states:`);
                data.forEach(state => {
                    terminalOutput('api', `  ${state.id}: ${state.name} (confidence: ${state.confidence.toFixed(2)})`);
                });
                
            } catch (error) {
                terminalOutput('error', `Failed to get states: ${error.message}`);
            }
        }

        async function evolveState() {
            if (!connected) {
                terminalOutput('error', 'Not connected to backend');
                return;
            }

            if (currentStates.length === 0) {
                terminalOutput('error', 'No states found. Click GET STATES first or initialize system.');
                return;
            }

            terminalOutput('system', `Evolving state ${currentStates[0].id}...`);
            
            try {
                const response = await fetch(`/api/states/${currentStates[0].id}/evolve`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json'
                    }
                });
                const data = await response.json();
                
                terminalOutput('success', 'State evolved successfully!');
                terminalOutput('api', `Parent: ${data.parent.id} - ${data.parent.name}`);
                terminalOutput('api', `Child: ${data.child.id} - ${data.child.name}`);
                terminalOutput('api', `Child confidence: ${data.child.confidence.toFixed(2)}`);
                terminalOutput('api', `Child swarm type: ${data.child.swarm_type}`);
                
                // Refresh states
                await getStates();
                
            } catch (error) {
                terminalOutput('error', `Failed to evolve state: ${error.message}`);
            }
        }

        // Auto-test connection on load
        window.onload = () => {
            terminalOutput('system', '=== MURPHY SYSTEM LIVE BACKEND DEMO ===');
            terminalOutput('system', 'Backend integrated on same server');
            terminalOutput('system', '');
            terminalOutput('system', 'Instructions:');
            terminalOutput('system', '1. Click TEST CONNECTION to verify backend');
            terminalOutput('system', '2. Click INITIALIZE SYSTEM to start Murphy System');
            terminalOutput('system', '3. Click GET STATUS to see system metrics');
            terminalOutput('system', '4. Click GET STATES to see all states');
            terminalOutput('system', '5. Click EVOLVE STATE to create child states');
            terminalOutput('system', '');
            
            // Auto-test after 2 seconds
            setTimeout(testConnection, 2000);
        };
    </script>
