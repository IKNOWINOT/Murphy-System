        const executeBtn = document.getElementById('executeBtn');

        executeBtn.addEventListener('click', executeCommand);
        commandInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') executeCommand();
        });

        function executeCommand() {
            const command = commandInput.value.trim().toLowerCase();
            
            if (command.includes('gap') || command.includes('analyze')) {
                document.querySelector('[data-tab="organization"]').click();
            } else if (command.includes('automat') || command.includes('recommend')) {
                document.querySelector('[data-tab="automation"]').click();
            } else if (command.includes('constraint') || command.includes('standard')) {
                document.querySelector('[data-tab="constraints"]').click();
            } else if (command === '/improve') {
                // Simulate improvement cycle
                alert('🔄 Running improvement cycle...\n\n' +
                      '1. Analyzing current gaps\n' +
                      '2. Generating automation recommendations\n' +
                      '3. Updating constraint matrix\n' +
                      '4. Optimizing production layer\n\n' +
                      'Improvement cycle complete! Check Automation tab for recommendations.');
                document.querySelector('[data-tab="automation"]').click();
