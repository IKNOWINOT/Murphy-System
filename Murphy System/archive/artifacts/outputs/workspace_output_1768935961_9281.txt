def create_document():
    """Create a living document"""
    data = request.json
    
    doc = runtime.create_living_document(
        title=data.get('title', 'Untitled'),
        content=data.get('content', ''),
        doc_type=data.get('type', 'general')
    )
    
    return jsonify(doc.to_dict())
