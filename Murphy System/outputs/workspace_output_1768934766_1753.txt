        async function executeMagnify() {
            const domain_name = document.getElementById('magnify-domain').value;
            
            addLog(`Magnifying document with ${domain_name} expertise...`, 'info');
            closeMagnifyModal();

            try {
                const response = await fetch(`${API_BASE}/api/documents/${currentDocId}/magnify`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ domain_name })  // Updated parameter name
                });

                const result = await response.json();
                
                if (result.success) {
                    addLog(`✓ Document magnified with ${domain_name} expertise`, 'success');
                    addLog(`✓ Expertise depth: ${result.expertise_depth}`, 'info');
                    document.getElementById('doc-state-display').textContent = 'Magnified';
                    document.getElementById('doc-content').value = result.content || document.getElementById('doc-content').value;
                } else {
                    addLog(`✗ Magnification failed: ${result.error || 'Unknown error'}`, 'error');
                }
            } catch (error) {
                addLog(`✗ Error: ${error.message}`, 'error');
            }
        }

        async function simplifyDocument() {
            if (!currentDocId) {
                addLog('✗ No document selected', 'error');
                return;
            }

            addLog('Simplifying document...', 'info');

            try {
                const response = await fetch(`${API_BASE}/api/documents/${currentDocId}/simplify`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' }
                });

                const result = await response.json();
                
                if (result.success) {
                    addLog('✓ Document simplified', 'success');
                    addLog(`✓ Expertise depth: ${result.expertise_depth}`, 'info');
                    document.getElementById('doc-state-display').textContent = 'Simplified';
                    document.getElementById('doc-content').value = result.content || document.getElementById('doc-content').value;
                } else {
                    addLog('✗ Simplification failed', 'error');
                }
            } catch (error) {
                addLog(`✗ Error: ${error.message}`, 'error');
            }
        }
