            // Add command to output
            const line = document.createElement('div');
            line.className = 'output-line';
            line.innerHTML = '<span class="output-prompt">&gt;&gt;</span> <span class="output-command">' + command + '</span>';
            output.appendChild(line);
            
            // Process command
            processCommand(command);
            
            // Scroll to bottom
            output.scrollTop = output.scrollHeight;
        }
        
        function processCommand(command) {
            const output = document.getElementById('output');
            const parts = command.split(' ');
            const cmd = parts[0].toLowerCase();
            const args = parts.slice(1);
            
            // Dynamic command processing
            if (cmd === 'help') {
                if (args.length > 0) {
                    showCommandHelp(args[0]);
                } else {
                    showAllCommands();
                }
            } else if (cmd === 'commands') {
                showAvailableCommands();
            } else if (cmd === 'status') {
                showSystemStatus();
            } else if (cmd === 'state') {
                processStateCommand(args);
            } else if (cmd === 'setup') {
                processSetupCommand(args);
            } else if (cmd === 'clear') {
                output.innerHTML = '';
            } else if (cmd === 'test') {
                runTest(args[0]);
            } else {
                // Check if it's a dynamic command
                if (availableCommands.includes(cmd)) {
                    executeDynamicCommand(cmd, args);
                } else {
                    addOutput('Unknown command: ' + cmd, 'error');
                }
            }
        }
        
        function showAllCommands() {
            addOutput('=== Available Commands ===', 'info');
            
            if (availableCommands.length === 0) {
                addOutput('No commands available. Couple modules to enable commands.', 'warning');
                return;
            }
            
            // Group by category
            const grouped = {};
            availableCommands.forEach(cmd => {
                const category = cmd.category || 'General';
                if (!grouped[category]) {
                    grouped[category] = [];
                }
                grouped[category].push(cmd);
            });
            
            for (const category in grouped) {
                addOutput('\n[' + category + ']', 'info');
                grouped[category].forEach(cmd => {
                    addOutput('  /' + cmd.name + ' - ' + cmd.description);
                });
            }
            
            addOutput('\nTotal: ' + availableCommands.length + ' commands', 'info');
        }
        
        function showCommandHelp(commandName) {
            const command = availableCommands.find(c => c.name === commandName);
            if (command) {
                addOutput('Command: /' + command.name, 'info');
                addOutput('Category: ' + command.category, 'info');
                addOutput('Description: ' + command.description, 'info');
                addOutput('Module: ' + command.module, 'info');
                addOutput('Risk Level: ' + command.riskLevel, 'info');
                
                if (command.parameters) {
                    addOutput('\nParameters:', 'info');
                    for (const param in command.parameters) {
                        addOutput('  ' + param + ': ' + command.parameters[param]);
                    }
                }
                
                if (command.examples) {
                    addOutput('\nExamples:', 'info');
                    command.examples.forEach(example => {
                        addOutput('  ' + example);
                    });
                }
            } else {
                addOutput('Command not found: ' + commandName, 'error');
            }
        }
        
        function processStateCommand(args) {
            if (args.length === 0) {
                addOutput('Usage: state [get|set|list|clear]', 'info');
                return;
            }
            
            const action = args[0].toLowerCase();
            
            if (action === 'get') {
                if (args.length < 2) {
                    showAllStates();
                } else {
                    const key = args[1];
                    const value = moduleStates[key];
                    if (value !== undefined) {
                        addOutput('State ' + key + ': ' + JSON.stringify(value), 'success');
                    } else {
                        addOutput('State not found: ' + key, 'warning');
                    }
                }
            } else if (action === 'set') {
                if (args.length < 3) {
                    addOutput('Usage: state set <key> <value>', 'info');
                    return;
                }
                const key = args[1];
                const value = args.slice(2).join(' ');
                moduleStates[key] = value;
                addOutput('State set: ' + key + ' = ' + value, 'success');
                saveStates();
            } else if (action === 'list') {
                showAllStates();
            } else if (action === 'clear') {
                moduleStates = {};
                addOutput('All states cleared', 'success');
                saveStates();
            } else {
                addOutput('Unknown state action: ' + action, 'error');
            }
        }
        
        function showAllStates() {
            addOutput('=== Current States ===', 'info');
            const keys = Object.keys(moduleStates);
            if (keys.length === 0) {
                addOutput('No states set', 'info');
            } else {
                keys.forEach(key => {
                    addOutput(key + ': ' + JSON.stringify(moduleStates[key]));
                });
            }
        }
        
        function processSetupCommand(args) {
            if (args.length === 0) {
                addOutput('Usage: setup [get|set|list]', 'info');
                return;
            }
            
            const action = args[0].toLowerCase();
            
            if (action === 'get') {
                showAllSetups();
            } else if (action === 'set') {
                if (args.length < 3) {
                    addOutput('Usage: setup set <category> <key> <value>', 'info');
                    return;
                }
                const category = args[1];
                const key = args[2];
                const value = args.slice(3).join(' ');
                
                if (!setupConfigs[category]) {
                    setupConfigs[category] = {};
                }
                setupConfigs[category][key] = value;
                addOutput('Setup updated: ' + category + '.' + key + ' = ' + value, 'success');
                saveSetups();
            } else if (action === 'list') {
                showAllSetups();
            } else {
                addOutput('Unknown setup action: ' + action, 'error');
            }
        }
        
        function showAllSetups() {
            addOutput('=== Setup Configurations ===', 'info');
            const categories = Object.keys(setupConfigs);
            if (categories.length === 0) {
                addOutput('No configurations set', 'info');

                    addOutput('\n[' + category + ']', 'info');
                    const config = setupConfigs[category];
                    for (const key in config) {
                        addOutput('  ' + key + ': ' + JSON.stringify(config[key]));
