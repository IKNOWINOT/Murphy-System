@app.route('/')
def serve_frontend():
    """Serve the frontend HTML page."""
    try:
        return send_file('/workspace/murphy_complete_v2.html')
    except Exception as e:
        logger.error(f"Error serving frontend: {e}")
        return jsonify({'success': False, 'error': str(e)}), 500
